// promptGuardInterceptor.js — Zero-trust middleware for Express.js
const crypto = require('crypto');
const sqlite3 = require('sqlite3').verbose();
const db = new sqlite3.Database('secure.db');

module.exports = function promptGuardInterceptor(req, res, next) {
  const { systemPrompt } = req.body;

  if (!systemPrompt) {
    return res.status(400).json({
      error: 'Missing systemPrompt in request body.'
    });
  }

  // SHA-256 hash the incoming prompt
  const hash = crypto.createHash('sha256')
    .update(systemPrompt).digest('hex');

  // Check if this hash was notarized by ubitquity1
  db.get(
    `SELECT * FROM prompt_logs WHERE hash = ?`, [hash],
    (err, row) => {
      if (!row) {
        console.warn(`BLOCKED: Unauthorized prompt. Hash: ${hash}`);
        return res.status(403).json({
          error: 'Security Violation: Prompt not notarized by ubitquity1.'
        });
      }
      next();  // Prompt is verified — pass through to LLM
    }
  );
};
// app.js — POST /api/verify endpoint
app.post('/api/verify', async (req, res) => {
  const { account, hash, txid, token } = req.body;

  if (!account || !hash || !txid || !token) {
    return res.status(400).json({
      error: 'Missing required fields'
    });
  }

  // Only ubitquity1 is authorized to notarize prompts
  if (account !== 'ubitquity1') {
    return res.status(403).json({
      error: 'Unauthorized account'
    });
  }

  const isValid = await verifyWebAuthTransaction(
    txid, hash, account, token
  );

  if (isValid) {
    db.run(
      `INSERT INTO prompt_logs (account, hash, txid, token_used)
       VALUES (?, ?, ?, ?)`,
      [account, hash, txid, token]
    );
    res.json({ success: true, message: 'Prompt notarized' });
  } else {
    res.status(400).json({
      error: 'Blockchain verification failed'
    });
  }
});
// app.js — Blockchain verification against XPR Network
async function verifyWebAuthTransaction(txid, expectedHash, expectedAccount, expectedToken) {
  try {
    const response = await fetch(
      'https://proton.cryptolions.io/v1/history/get_transaction',
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ id: txid })
      }
    );

    const data = await response.json();
    const action = data.trx.trx.actions[0];

    // Verify contract, action, sender, quantity & memo
    const isValid =
      action.account === 'tokencreate' &&
      action.name    === 'transfer' &&
      action.data.from     === expectedAccount &&
      action.data.quantity === `1.00000000 ${expectedToken}` &&
      action.data.memo     === `REPORT:${expectedHash}:CLEAN_PROMPT`;

    return isValid;
  } catch (err) {
    console.error('Verification error:', err);
    return false;
  }
}
